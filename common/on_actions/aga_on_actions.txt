			    limit = {
			        any_country = { axis = yes has_tech = survivability_studies }
			    }

			        limit = {
			            axis = yes
			        }
			        set_technology = {
			            survivability_studies = 1
			        }

		}

				limit = {

						rule = argentina_existence

					}
				}

            	    limit = {
            	        is_argentina = yes
            	    }

            	    add_core_of = ARG
            	}
			}
			else = {

           	}

				add_opinion_modifier = { target = BUL modifier = huge_increase  }
				}

				limit = {
					is_playable_country = yes

				army_experience = 9
