on_actions = {
	on_monthly = {
		effect = {
			if = {
				limit = {
					any_country = { axis = yes has_tech = antitank2 }
				}
				axis = { scoped_play_song = "entertheeast" }			
			}
		}
	}
	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					NOT = { ENG = { controls_state = 907 } }
				}
				build_railway = {
					level = 1
					path = { 12725 2003 2046 }
				}
			}
		}
	}	
	on_startup = {
		effect = {
			FIN = {
				add_opinion_modifier = { target = YUG modifier = large_decrease }
				add_opinion_modifier = { target = YUG modifier = hostile_status }
				add_opinion_modifier = { target = ALB modifier = large_decrease }
				add_opinion_modifier = { target = ALB modifier = hostile_status }
				add_opinion_modifier = { target = GRE modifier = large_decrease }
				add_opinion_modifier = { target = GRE modifier = hostile_status }
				add_opinion_modifier = { target = ITA modifier = huge_increase  }
				add_opinion_modifier = { target = BUL modifier = huge_increase  }
				}
			every_country = {
				limit = {
					is_playable_country = yes
				}
				air_experience = 10
				army_experience = 10
				VEN = {
					country_event = {
					id = AGA.1
					days = 30 
					}
				}				
			}
			if = { 
				limit = { 
					AND = { 
						AST = { is_ai = yes } 
						SAF = { is_ai = yes } 
						CAN = { is_ai = yes } 
						RAJ = { is_ai = yes } 
						NZL = { is_ai = yes } 
					}
				}
				ENG = { add_research_slot = 1 }
			}
		}
	}
	on_declare_war = {
		effect = {
			if = {
				limit = {
					FROM = {
						tag = ENG
					}
					ENG = { has_war = yes }
					date>1941.1.1	
				}
				JAP = { add_timed_idea = { idea = JAP_tora_tora_tora_idea days = 180 } }
				JAP = { declare_war_on = { target = USA type = annex_everything } }
				JAP = { declare_war_on = { target = FRA type = annex_everything } }
				JAP = { declare_war_on = { target = SAF type = annex_everything } }
				JAP = { declare_war_on = { target = CAN type = annex_everything } }
				JAP = { declare_war_on = { target = AST type = annex_everything } }
				JAP = { declare_war_on = { target = NZL type = annex_everything } }
				JAP = { declare_war_on = { target = HOL type = annex_everything } }
				JAP = { declare_war_on = { target = BEL type = annex_everything } }
				if = { limit = { BRA = { is_ai = no } } JAP = { declare_war_on = { target = BRA type = annex_everything } } }
				if = { limit = { MEX = { is_ai = no } } JAP = { declare_war_on = { target = MEX type = annex_everything } } }
			}
		}
	}		
}
