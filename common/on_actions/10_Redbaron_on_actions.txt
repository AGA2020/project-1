on_actions = {

	on_startup = { 
		effect = {
			set_global_flag = { flag = VIC value = 0 }
			set_variable = {
				global.axis_score = 0
			}
			set_variable = {
				global.allies_score = 0
			}

		}
		effect = {
			every_country = {
				limit = {
					is_playable_country = yes
				}
				add_ideas = {
					victory_con
				}
			}

		}

	}

	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					FROM.FROM = {
						OR = {
							state = 613 #Nanjing
							state = 439 #Dehli
							state = 447 #Alexandria
							state = 450 #Benghasi
							state = 7 #Amsterdam
							state = 16 #Paris
							state = 20 #Lyon
							state = 224 #Bryansk
							state = 219 #Moscow
							state = 217 #Stalingrad
						}
					}
				}
				ROOT = { #gained controll (add var)
					if = {
						limit = {
							is_allied_country = yes
						}
						add_to_variable = {
							var = global.allies_score
							value = 1
						}
						clamp_variable = {
							var = global.allies_score
							min = 0
							max = 10
						}
					}
					else_if = {
						limit = { 
							is_axis_country = yes
						}
						add_to_variable = {
							var = global.axis_score
							value = 1
						}
						clamp_variable = {
							var = global.axis_score
							min = 0
							max = 10
						}
					}
				}
				FROM = { #lost controll (sub var)
					if = {
						limit = {
							is_allied_country = yes
						}
						subtract_from_variable = {
							var = global.allies_score
							value = 1
						}
						clamp_variable = {
							var = global.allies_score
							min = 0
							max = 10
						}
					}
					else_if = {
						limit = { 
							is_axis_country = yes
						}
						subtract_from_variable = {
							var = global.axis_score
							value = 1
						}
						clamp_variable = {
							var = global.axis_score
							min = 0
							max = 10
						}
					}
				}
			}
		}
	}
	
	
	
	
}
